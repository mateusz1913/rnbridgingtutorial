### Boolean argument

<Tabs groupId="specs">

import type { TurboModule } from 'react-native';
import { TurboModuleRegistry } from 'react-native';

//

export interface Spec extends TurboModule {
  myMethod(enabled: boolean, checked?: boolean): void
